```python
def change_tp(cell, tp: int):
    """ Преобразует значение ТП в зависимости от выбора ТП"""
    if tp == 1:
        if cell in [2, 3]:
            return 1
        else:
            return cell
    elif tp == 2:
        if cell == 3:
            return 2
        elif cell == 1:
            return -1
        else:
            return cell
    elif tp == 3:
        if cell in [1, 2]:
            return -1
        else:
            return cell
    else:
        return cell
    
    
TP = 1  # Выбираем ТП=СЛ

# TP = 3  # Выбираем 3*ТП=СЛ
```
